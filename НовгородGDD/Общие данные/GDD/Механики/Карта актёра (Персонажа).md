using System;

[Flags]

public enum NPCPersonality

{

    None = 0,

    Greedy = 0b0000001,      // Жадный

    Generous = 0b0000011,    // Щедрый

    Ambitious = 0b0000100,   // Честолюбивый

    Honorable = 0b0001000,   // Честный

    Devious = 0b0011000,     // Коварный

    Loyal = 1 << 4,          // Лояльный

    Treacherous = 1 << 5,    // Предательский

    Pious = 1 << 6,          // Благочестивый

    Pragmatic = 1 << 7,      // Прагматичный

    HotTempered = 1 << 8,    // Вспыльчивый

    Cautious = 1 << 9       // Осторожный

}

  

public class Person

{

    public string Name { get; set; }

    public NPCPersonality Personality { get; set; }

    public List<Relation> Acquaintances { get; set; } = new List<Relation>();

    {

        return StoryGenerator.GenerateEventStory(this);

    }

}

  

public class Relation

    public static string GenerateEventStory(Person npc)

    {

        // Выбираем случайное действие из MyActions или PersonActions

        ActionPlayer action = npc.Acquaintances.SelectMany(r => r.MyActions).FirstOrDefault()

            ?? new ActionPlayer

            {

                Description = "потребовал поддержки",

                PersonTarget = npc.Acquaintances.FirstOrDefault()?.Person ?? new Person { Name = "некто" },

                TypeOfLocation = 1,

                Reason = "чтобы укрепить власть",

                InputAction = new InputActionPlayer { Description = "согласился", Reason = "ради выгоды" }

            };

  

        // Выбираем случайную черту

        var traits = Enum.GetValues(typeof(NPCPersonality))

            .Cast<NPCPersonality>()

            .Where(t => t != NPCPersonality.None && npc.Personality.HasFlag(t))

            .Take(3)

            .ToList();

        if (!traits.Any()) traits.Add(NPCPersonality.Pragmatic);

        var trait = traits[Random.Range(0, traits.Count)];

  

        // Формируем историю

        string intro = Intros.ContainsKey(trait) ? Intros[trait][Random.Range(0, Intros[trait].Length)] : "Однажды";

        string initiator = npc.Name;

        string location = Locations.ContainsKey(action.TypeOfLocation) ? Locations[action.TypeOfLocation] : "где-то";

        string eventDesc = action.Description;

        string target = action.PersonTarget?.Name ?? "некто";

        string reason = action.Reason;

        string consequence = action.InputAction?.Description ?? Outcomes[trait][Random.Range(0, 2)];

  

        return $"{intro}, {initiator} {location} {eventDesc} по отношению к {target}, {reason}. {consequence}.";

    }

  

    public static string GenerateAgreementStory(Person npc)

    {

        // Выбираем случайную договорённость

        MutualAgreement agreement = npc.Acquaintances.SelectMany(r => r.MutualAgreements).FirstOrDefault()

            ?? new MutualAgreement

            {

                Description = "союз против врага",

                Success = true,

                Expired = false,

                IsOfficial = true

            };

  

        // Выбираем случайную связь

        Relation relation = npc.Acquaintances.FirstOrDefault() ?? new Relation(new Person { Name = "некто" }, 0.5f);

  

        // Выбираем случайную черту

        var traits = Enum.GetValues(typeof(NPCPersonality))

            .Cast<NPCPersonality>()

            .Where(t => t != NPCPersonality.None && npc.Personality.HasFlag(t))

            .Take(3)

            .ToList();

        if (!traits.Any()) traits.Add(NPCPersonality.Pragmatic);

        var trait = traits[Random.Range(0, traits.Count)];

  

        // Формируем историю

        string location = Locations.ContainsKey(Random.Range(1, 6)) ? Locations[Random.Range(1, 6)] : "где-то";

        string parties = $"между мной и {relation.Person.Name}";

        string agreementDesc = agreement.IsOfficial ? $"официально заключили {agreement.Description}" : $"тайно договорились о {agreement.Description}";

        string attitude = agreement.Success ? Outcomes[trait][0] : Outcomes[trait][1];

  

        return $"{location}, {parties} {agreementDesc}. {attitude}.";

    }

}
